pipeline {
    agent any

     environment {
        AWS_ACCESS_KEY_ID     = credentials('AWS_ACCESS_KEY_ID_DEV')
        AWS_SECRET_ACCESS_KEY = credentials('AWS_SECRET_ACCESS_KEY_DEV')
    }
    
    stages {
        stage('checkout') {
            steps {
		checkout([$class: 'GitSCM', branches: [[name: '*/release-branch.v1.dev']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/juanmanuel0963/terraform_jenkins_api_gateway_lambda_func']]])
            }
        }
        stage('Terraform init') {
            steps {
                sh ("terraform init");
            }
            
        }
        stage('Terraform action') {
            steps {
                echo "terraform action from the parameter is --> ${action}"
                sh ("terraform ${action} --auto-approve");
            }
            
        }        
    }
}
